Class FiguraGeometrica{
	protected 
		String color;

	public

		metodo obt_color(){
			return color;
		}

		FiguraGeometrica(String _color){
			color = _color;
		}

		metodo leer(){
			read(color);
		}

		metodo mostrar(){
			print(color);
		}
}


Class Triangulo extends FiguraGeometrica{
	private 
		double lado_1, lado_2, lado_3;

	public

		Triangulo(String _color, double l1, double l2, double l3){
			super(_color);
			lado_1 = l1;
			lado_2 = l2;
			lado_3 = l3;
		}

		metodo obt_lado_1(){
			return lado_1;
		}

		metodo obt_lado_2(){
			return lado_2;
		}

		metodo obt_lado_3(){
			return lado_3;
		}

		metodo leer(){
			super.leer();
			read(lado_1, lado_2, lado_3);
		}


		metodo mostrar(){
			super.mostrar();
			print(lado_1, lado_2, lado_3);
		}


		metodo area() {
			double s = (lado_1 + lado_2 + lado_3) / 2;
			double area = Math.sqrt(s * (s - lado_1) * (s - lado_2) * (s - lado_3));
			return area;
		}

		metodo isoceles(){
			if(lado_1 == lado_2 || lado_1 == lado_3 || lado_2 == lado_3) then {
				print("es isoceles");
			}else{
				print("no es isoceles");
			}
		}

		metodo rectangulo() {
			double c = 0, a = 0, b = 0;
			if(lado_1 > c) then {
				c = lado_1;
				a = lado_2;
				b = lado_3;

		}
}

		}
